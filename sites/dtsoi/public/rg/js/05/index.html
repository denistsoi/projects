(answer)

let counter = 0;

function add() {
	counter = counter + parseInt(userInput.value);
}

function minus() {
	counter = counter - parseInt(userInput.value);
}

function multiply() {
	counter = counter * parseInt(userInput.value);
}

function divide() {
	counter = counter / parseInt(userInput.value);
}
